分块入门

2021-03-13 题解

分块

所谓数列分块，就是将一个数列分成若干块~~（废话）~~，通过块的整体操作和边界的局部暴力，达到优化时间复杂度的问题。通过LOJ的数列分块入门系列，我们对数列分块系列进行一定的了解。

数列分块入门 1

要求实现区间加值，单点查询。

将整个数列分为n\sqrt{n}n​块，对于每个块单独记录总和。首先考虑修改，那么我们就可以把从lll到rrr的大区间修改分成若干个块，对于两端不是整块的部分，直接暴力修改；为了实现整块的快速修改，我们还需要定义一个plsplspls数组，pls[i]pls[i]pls[i]代表第iii个块整体加的值 ,这样整块就可以做到O(1)O(1)O(1)修改，查询时只要加上所在块的plsplspls即可。分块第一题，建议好好理解，体会分块的思想。

#include<bits/stdc++.h>
#define N 50010
using namespace std;
int n,a[N],pls[N],pos[N],L[N],R[N];
//L,R记录每个块的左右边界,pos存储每个数所在的块的编号
void build() { //分块
	int t=sqrt(n);
	for (int i=1;i<=t;i++) { //记录左右边界
		L[i]=(i-1)*t+1;
		R[i]=i*t;
	}
	if (R[t]<n) L[++t]=R[t-1]+1,R[t]=n; //处理分块的多余部分
	for (int i=1;i<=t;i++) {
		for (int j=L[i];j<=R[i];j++) pos[j]=i; //处理pos数组
	}
}
void add(int l,int r,int v) { //区间加值
	int p=pos[l],q=pos[r]; 
	if (p==q) { //如果两数在同一块暴力处理
		for (int i=l;i<=r;i++) a[i]+=v;
		return;
	}
	for (int i=p+1;i<=q-1;i++) pls[i]+=v; //整块处理
	for (int i=l;i<=R[p];i++) a[i]+=v; //边界暴力处理
	for (int i=L[q];i<=r;i++) a[i]+=v;
}
int ask(int x) {
	return a[x]+pls[pos[x]];
}
signed main() {
	scanf("%d",&n);
	for (int i=1;i<=n;i++) {
		scanf("%d",&a[i]);
	}
	build();
	for (int i=1;i<=n;i++) {
		int opt,l,r,c;
		scanf("%d%d%d%d",&opt,&l,&r,&c);
		if (opt==0) add(l,r,c);
		else printf("%d\n",ask(r));
	}
}

数列分块入门4

区间加值，区间求和。

为什么现讲4，因为4的思想和1更为接近。

同样运用1的思想，在查询时与修改一样，在两端暴力查询，中间直接运用sumsumsum和plsplspls快速求和。

int ask(int l,int r,int ans=0) {
	int p=pos[l],q=pos[r];
	if (p==q) {
		for (int i=l;i<=r;i++) ans+=a[i];
		return ans+pls[p]*(r-l+1);
	}
	for (int i=p+1;i<=q-1;i++) ans+=sum[i]+pls[i]*(R[i]-L[i]+1);
	for (int i=l;i<=R[p];i++) ans+=a[i];
	ans+=pls[p]*(R[p]-l+1);
	for (int i=L[q];i<=r;i++) ans+=a[i];
	ans+=pls[q]*(r-L[q]+1);
	return ans;
}

数列分块入门2

区间加值，查询区间内小于某值的元素个数。

分块后，如何快速获取小于某值的元素个数？不难想到，若序列有序，那么通过二分查找即可实现快速查询。所以我们将每组的元素放入一个vector并排序，在边界暴力修改时，只需要重新构建vector，即所谓的重构，就可以维护每个块有序。

#include<bits/stdc++.h>
#define N 500010
#define int long long
using namespace std;
int n,L[N],R[N],a[N],pls[N],pos[N];
vector<int> v[N];
void build() {
	int t=sqrt(n);
	for (int i=1;i<=t;i++) {
		L[i]=(i-1)*t+1;
		R[i]=i*t;
	}
	if (R[t]<n) t++,L[t]=R[t-1]+1,R[t]=n;
	for (int i=1;i<=t;i++) {
		for (int j=L[i];j<=R[i];j++) {
			pos[j]=i;
			v[i].push_back(a[j]);
		}
		sort(v[i].begin(),v[i].end());
	}
}
void change(int id,int l,int r,int x) { //暴力修改并重构
	for (int i=l;i<=r;i++) a[i]+=x;
	v[id].clear();
	for (int i=L[id];i<=R[id];i++)
		v[id].push_back(a[i]);
	sort(v[id].begin(),v[id].end());
}
void add(int l,int r,int x) {
	int p=pos[l],q=pos[r];
	if (p==q) {change(p,l,r,x);return;}
	for (int i=p+1;i<=q-1;i++) pls[i]+=x;
	change(p,l,R[p],x),change(q,L[q],r,x);
}
int ask(int l,int r,int x,int ans=0) {
	int p=pos[l],q=pos[r];
	if (p==q) {
		for (int i=l;i<=r;i++) if (a[i]+pls[p]<x) ans++;
		return ans;
	}
	for (int i=p+1;i<=q-1;i++)
		ans+=(lower_bound(v[i].begin(),v[i].end(),x-pls[i])-v[i].begin());
	     /* lower_bound是std自带的二分,与sort用法类似，复杂度O(logn),返回第一个
	     大于等于查询值的地址，减去开头地址即为下标，再减一就是第一个小于查询值的地址 */
	for (int i=l;i<=R[p];i++) if (a[i]+pls[p]<x) ans++;
	for (int i=L[q];i<=r;i++) if (a[i]+pls[q]<x) ans++;
	return ans;
}
signed main() {
	scanf("%lld",&n);
	for (int i=1;i<=n;i++) {
		scanf("%lld",&a[i]);
	}
	build();
	for (int i=1;i<=n;i++) {
		int opt=0,l=0,r=0,c=0;
		scanf("%lld%lld%lld%lld",&opt,&l,&r,&c);
		if (!opt) add(l,r,c);
		else printf("%lld\n",ask(l,r,c*c));
	}
}

数列分块入门 3

区间加值，查询区间内小于某值的最大元素值。

与2基本相同，只将个数改为最大值即可。

#include<bits/stdc++.h>
#define N 500010
#define int long long
#define Max(x,y) (x>y?x:y)
using namespace std;
int n,L[N],R[N],a[N],pls[N],pos[N];
vector<int> v[N];
void build() {
	int t=sqrt(n);
	for (int i=1;i<=t;i++) {
		L[i]=(i-1)*t+1;
		R[i]=i*t;
	}
	if (R[t]<n) t++,L[t]=R[t-1]+1,R[t]=n;
	for (int i=1;i<=t;i++) {
		for (int j=L[i];j<=R[i];j++) {
			pos[j]=i;
			v[i].push_back(a[j]);
		}
		sort(v[i].begin(),v[i].end());
	}
}
void change(int id,int l,int r,int x) {
	for (int i=l;i<=r;i++) a[i]+=x;
	v[id].clear();
	for (int i=L[id];i<=R[id];i++)
		v[id].push_back(a[i]);
	sort(v[id].begin(),v[id].end());
}
void add(int l,int r,int x) {
	int p=pos[l],q=pos[r];
	if (p==q) {change(p,l,r,x);return;}
	for (int i=p+1;i<=q-1;i++) pls[i]+=x;
	change(p,l,R[p],x),change(q,L[q],r,x);
}
int ask(int l,int r,int x,int ans=-1) {
	int p=pos[l],q=pos[r];
	if (p==q) {
		for (int i=l;i<=r;i++) if (a[i]+pls[p]<x) ans=Max(ans,a[i]+pls[p]);
		return ans;
	}
	for (int i=p+1;i<=q-1;i++) {
		int j=lower_bound(v[i].begin(),v[i].end(),x-pls[i])-v[i].begin();
		if (j!=0) ans=Max(ans,v[i][j-1]+pls[i]);
	}
	for (int i=l;i<=R[p];i++) if (a[i]+pls[p]<x) ans=Max(ans,a[i]+pls[p]);
	for (int i=L[q];i<=r;i++) if (a[i]+pls[q]<x) ans=Max(ans,a[i]+pls[q]);
	return ans;
}
signed main() {
	scanf("%lld",&n);
	for (int i=1;i<=n;i++) {
		scanf("%lld",&a[i]);
	}
	build();
	for (int i=1;i<=n;i++) {
		int opt=0,l=0,r=0,c=0;
		scanf("%lld%lld%lld%lld",&opt,&l,&r,&c);
		if (!opt) add(l,r,c);
		else printf("%lld\n",ask(l,r,c));
	}
}

数列分块入门5

区间开方，区间求和

乍一看似乎无从下手，开方似乎不可能实现区间整体维护。但再仔细思考一下，不难发现开方运算导致数字值的下降非常快，几次的操作就可以变为000或111，这时候再开方数字不会变化，想通这个这道题也就迎刃而解了。用vis[i]vis[i]vis[i]数组记录第iii个块每个数是否都是000或111。每次进行修改时若vis[i]=0vis[i]=0vis[i]=0则暴力修改，否则不用再修改。

#include<bits/stdc++.h>
#define N 50010
#define int long long
using namespace std;
int n,a[N],pls[N],pos[N],L[N],R[N],sum[N],vis[N];
void build() {
	int t=sqrt(n);
	for (int i=1;i<=t;i++) {
		L[i]=(i-1)*t+1;
		R[i]=i*t;
	}
	if (R[t]<n) t++,L[t]=R[t-1]+1,R[t]=n;
	for (int i=1;i<=t;i++) {
		for (int j=L[i];j<=R[i];j++)
			pos[j]=i,sum[i]+=a[j];
	}
}
void change(int id,int l,int r) {
	if (vis[id]) return; vis[id]=1;
	for (int i=l;i<=r;i++) a[i]=sqrt(a[i]);
	sum[id]=0;
	for (int i=L[id];i<=R[id];i++) {
		sum[id]+=a[i];
		if (a[i]>1) vis[id]=0;
	}
}
void add(int l,int r) {
	int p=pos[l],q=pos[r];
	if (p==q) {change(p,l,r);return;}
	for (int i=p+1;i<=q-1;i++) change(i,L[i],R[i]);
	change(p,l,R[p]); change(q,L[q],r);
}
int ask(int l,int r,int ans=0) {
	int p=pos[l],q=pos[r];
	if (p==q) {
		for (int i=l;i<=r;i++) ans+=a[i];
		return ans;
	}
	for (int i=p+1;i<=q-1;i++) ans+=sum[i];
	for (int i=l;i<=R[p];i++) ans+=a[i];
	for (int i=L[q];i<=r;i++) ans+=a[i];
	return ans;
}
signed main() {
	scanf("%lld",&n);
	for (int i=1;i<=n;i++) {
		scanf("%lld",&a[i]);
	}
	build();
	for (int i=1;i<=n;i++) {
		int opt,l,r,c;
		scanf("%lld%lld%lld%lld",&opt,&l,&r,&c);
		if (opt==0) add(l,r);
		else printf("%lld\n",ask(l,r));
	}
}
